<objective>
Wire the verifier agent to validate Flutter UI TRDs at the artifact level. This is the third enforcement gate (after planner 10-03 and executor 10-04). The verifier must:

1. **State coverage grep (REQ-10-05 primary):** For each `type: ui` artifact, load the regex catalog from `references/flutter-state-patterns.md` matching the TRD's `state_management:`. For each declared state in `must_haves.artifacts[*].states[]`, grep the artifact's `tests.widget` file with the corresponding regex(es). HIGH-confidence patterns: block on miss. MEDIUM/LOW: advise on miss.

2. **API contract drift step (REQ-10-05 secondary):** Invoke `df-tools verify api-contract <trd-path>` (already shipped by TRD 10-08); record drift entries in VERIFICATION.md's `drift:` section as advisory. Never blocks.

3. **Semantics() static scan (REQ-10-05 advisory):** Grep widget tests + source files for `Semantics(`. If a `type: ui` artifact has zero Semantics() wrappers, emit advisory note ("missing Semantics() likely breaks Maestro selectability").

4. **Maestro orphan-flow detection:** Cross-reference `.maestro/*.yaml` flow files against the union of `must_haves.artifacts[*].tests.maestro` across all TRDs in the objective. Flag orphan flows as advisory (legitimate orphans like `app_launch.yaml` are expected).

Purpose: Without this gate, the planner enforces schema and the executor enforces test runs — but nothing checks that declared `states:` are ACTUALLY covered by the widget test, that referenced contract files haven't drifted, or that Semantics() wrappers exist for Maestro selectors. This TRD closes the verification loop.

Output: New lib `flutter-state-coverage.cjs` (~80 lines) + paired test (~120 lines) + df-tools subcommand + verifier.md extensions (~60 lines added).

Why TDD: state-coverage verification is pure logic (regex map + grep result → status decision). Every confidence-tier × library combination gets a RED-GREEN cycle.

Create `flutter-state-coverage.cjs`:

```javascript
'use strict';
const fs = require('fs');
const path = require('path');
const { output } = require('./helpers.cjs');

const LIBRARY_HEADINGS = {
  '## Riverpod': 'riverpod',
  '## flutter_bloc': 'bloc',
  '## setState': 'setState',
  '## Other': 'other',
};

/**
 * Parse references/flutter-state-patterns.md into a structured catalog.
 * Catalog shape: { riverpod: [{name, pattern, covers, confidence, note}, ...], bloc: [...], setState: [...] }
 */
function loadCatalog(catalogPath) {
  const content = fs.readFileSync(catalogPath, 'utf-8');
  const catalog = { riverpod: [], bloc: [], setState: [] };
  const sections = content.split(/^##\s+/m);
  for (const section of sections) {
    const firstLine = section.split('\n')[0];
    let lib = null;
    for (const [heading, key] of Object.entries(LIBRARY_HEADINGS)) {
      // Strip "## " prefix from heading for matching
      const headingText = heading.replace(/^##\s+/, '');
      if (firstLine.startsWith(headingText)) {
        lib = key;
        break;
      }
    }
    if (!lib || lib === 'other') continue;

    // Extract YAML block — the first ```yaml ... ``` after the heading
    const yamlMatch = section.match(/```yaml\n([\s\S]+?)\n```/);
    if (!yamlMatch) continue;

    catalog[lib] = parseSimpleYamlList(yamlMatch[1]);
  }
  return catalog;
}

/**
 * Tiny YAML-list-of-objects parser for the catalog shape:
 *   - name: foo
 *     pattern: '...'
 *     covers: [a, b]
 *     confidence: HIGH
 *     note: "..."
 */
function parseSimpleYamlList(yamlBody) {
  const entries = [];
  const lines = yamlBody.split('\n');
  let current = null;
  for (const line of lines) {
    if (/^\s*-\s+name:\s*(.+)/.test(line)) {
      if (current) entries.push(current);
      current = { name: RegExp.$1.trim() };
    } else if (current) {
      const kvMatch = line.match(/^\s+(\w+):\s*(.+)/);
      if (!kvMatch) continue;
      const [, key, rawValue] = kvMatch;
      let value = rawValue.trim();
      // Strip quotes
      if ((value.startsWith("'") && value.endsWith("'")) || (value.startsWith('"') && value.endsWith('"'))) {
        value = value.slice(1, -1);
      }
      // Parse inline arrays: [a, b, c]
      if (value.startsWith('[') && value.endsWith(']')) {
        value = value.slice(1, -1).split(',').map(s => s.trim()).filter(Boolean);
      }
      current[key] = value;
    }
  }
  if (current) entries.push(current);
  return entries;
}

/**
 * Verify coverage of declared states against widget test content using the catalog.
 *
 * Returns:
 *   { status: 'verified' | 'partial' | 'missing' | 'skipped',
 *     coverage: { <state>: { matched: bool, matchedBy: string|null, confidence: string|null } },
 *     blockers: [<state>],     // HIGH-confidence misses
 *     advisories: [<msg>] }
 */
function verifyCoverage({ stateManagement, declaredStates, widgetTestContent, catalog }) {
  if (stateManagement === 'other' || !catalog[stateManagement]) {
    return { status: 'skipped', coverage: {}, blockers: [], advisories: ['regex skipped for state_management:' + stateManagement] };
  }
  const patterns = catalog[stateManagement];
  const coverage = {};
  const blockers = [];
  const advisories = [];

  for (const state of declaredStates) {
    const relevant = patterns.filter(p => p.covers && p.covers.includes(state));
    let matched = false;
    let matchedBy = null;
    let highestConf = null;
    for (const p of relevant) {
      try {
        const re = new RegExp(p.pattern);
        if (re.test(widgetTestContent)) {
          matched = true;
          matchedBy = p.name;
          highestConf = p.confidence;
          if (p.confidence === 'HIGH') break;  // short-circuit on HIGH
        }
      } catch (e) {
        advisories.push(`bad regex in catalog for ${p.name}: ${e.message}`);
      }
    }
    coverage[state] = { matched, matchedBy, confidence: highestConf };
    if (!matched) {
      // Determine if any HIGH pattern was expected for this state
      const hasHighPattern = relevant.some(p => p.confidence === 'HIGH');
      if (hasHighPattern) {
        blockers.push(state);
      } else if (relevant.length > 0) {
        advisories.push(`state '${state}' has only MEDIUM/LOW-confidence patterns; manual review recommended`);
      } else {
        advisories.push(`state '${state}' has no catalog pattern; manual review recommended`);
      }
    }
  }

  let status;
  if (blockers.length === declaredStates.length) status = 'missing';
  else if (blockers.length > 0) status = 'partial';
  else status = 'verified';

  return { status, coverage, blockers, advisories };
}

/** df-tools subcommand handler (Task 2 wires this in) */
function cmdVerifyFlutterStateCoverage(cwd, trdPath, raw) {
  if (!trdPath) {
    output({ error: 'TRD path required' }, raw);
    return;
  }
  const { extractFrontmatter } = require('./frontmatter.cjs');
  const absTrd = path.isAbsolute(trdPath) ? trdPath : path.join(cwd, trdPath);
  if (!fs.existsSync(absTrd)) {
    output({ error: 'TRD not found', trd_path: trdPath }, raw);
    return;
  }
  const trdContent = fs.readFileSync(absTrd, 'utf-8');
  const fm = extractFrontmatter(trdContent);
  if (fm.type !== 'ui' || fm.stack !== 'flutter') {
    output({ status: 'not_applicable', reason: 'not a Flutter UI TRD', trd_path: absTrd }, raw);
    return;
  }
  const catalogPath = path.join(__dirname, '..', '..', 'references', 'flutter-state-patterns.md');
  const catalog = loadCatalog(catalogPath);
  const artifacts = (fm.must_haves && fm.must_haves.artifacts) || [];
  const perArtifact = [];
  for (const art of artifacts) {
    if (!art.states || !art.tests || !art.tests.widget) continue;
    const widgetPath = path.isAbsolute(art.tests.widget) ? art.tests.widget : path.join(cwd, art.tests.widget);
    if (!fs.existsSync(widgetPath)) {
      perArtifact.push({ artifact: art.path, status: 'missing_test_file', test_path: art.tests.widget });
      continue;
    }
    const widgetTestContent = fs.readFileSync(widgetPath, 'utf-8');
    const result = verifyCoverage({
      stateManagement: fm.state_management,
      declaredStates: art.states,
      widgetTestContent,
      catalog,
    });
    perArtifact.push({ artifact: art.path, ...result });
  }
  const aggregate = {
    trd_path: absTrd,
    state_management: fm.state_management,
    artifacts: perArtifact,
    overall: perArtifact.every(a => a.status === 'verified') ? 'verified' : 'partial',
  };
  output(aggregate, raw);
}
```
